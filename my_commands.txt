# Build container
podman build -f Dockerfile.explainable-mortal -t explainable-mortal .

# Basic usage (shows help)
podman run --rm mortal-simple

# Run training_grp
cd mortal
podman run --rm \
   -v .:/workspace/mortal \
   -w /workspace/mortal \
   --entrypoint python \
   localhost/explainable-mortal \
   train.py

# Tensorboard
cd mortal
podman run --rm -it \
   -p 6006:6006 \
   -v .:/workspace/mortal \
   -w /workspace/mortal \
   --entrypoint tensorboard \
   localhost/explainable-mortal \
   --logdir=/workspace/mortal/built/tensorboard --host=0.0.0.0
