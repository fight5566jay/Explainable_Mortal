# Build container
podman build -f Dockerfile.explainable-mortal -t explainable-mortal .

# Basic usage (shows help)
podman run --rm localhost/explainable-mortal

# Run training (offline)

   -v .:/workspace/mortal \
   -w /workspace/mortal \

   train.py
